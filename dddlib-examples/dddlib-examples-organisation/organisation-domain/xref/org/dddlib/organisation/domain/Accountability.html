1   package org.dddlib.organisation.domain;
2   
3   import java.util.Date;
4   import java.util.List;
5   
6   import javax.persistence.*;
7   
8   import org.dayatang.domain.AbstractEntity;
9   import org.dayatang.utils.DateUtils;
10  
11  @Entity
12  @Table(name = "accountabilities")
13  @Inheritance(strategy = InheritanceType.SINGLE_TABLE)
14  @DiscriminatorColumn(name = "CATEGORY", discriminatorType = DiscriminatorType.STRING)
15  @NamedNativeQueries({
16      @NamedNativeQuery(name = "Accountability.findAccountabilitiesByParty",
17              query = "select o from Accountability o where (o.commissioner = :party or o.responsible = :party) and o.fromDate <= :date and o.toDate > :date")})
18  public abstract class Accountability<C extends Party, R extends Party> extends AbstractEntity {
19  
20      private static final long serialVersionUID = 3456398163374995470L;
21  
22      @ManyToOne(targetEntity = Party.class, cascade = CascadeType.ALL)
23      @JoinColumn(name = "commissioner_id")
24      private C commissioner;
25  
26      @ManyToOne(targetEntity = Party.class, cascade = CascadeType.ALL)
27      @JoinColumn(name = "responsible_id")
28      private R responsible;
29  
30      @Temporal(TemporalType.DATE)
31      @Column(name = "from_date")
32      private Date fromDate;
33  
34      @Temporal(TemporalType.DATE)
35      @Column(name = "to_date")
36      private Date toDate;
37  
38      protected Accountability() {
39      }
40  
41      public Accountability(C commissioner, R responsible, Date fromDate) {
42          this.commissioner = commissioner;
43          this.responsible = responsible;
44          this.fromDate = new Date(fromDate.getTime());
45          this.toDate = DateUtils.MAX_DATE;
46      }
47  
48      public C getCommissioner() {
49          return commissioner;
50      }
51  
52      public void setCommissioner(C commissioner) {
53          this.commissioner = commissioner;
54      }
55  
56      public R getResponsible() {
57          return responsible;
58      }
59  
60      public void setResponsible(R responsible) {
61          this.responsible = responsible;
62      }
63  
64      public Date getFromDate() {
65          return new Date(fromDate.getTime());
66      }
67  
68      void setFromDate(Date fromDate) {
69          this.fromDate = fromDate;
70      }
71  
72      public Date getToDate() {
73          return toDate;
74      }
75  
76      void setToDate(Date toDate) {
77          this.toDate = toDate;
78      }
79  
80      @Override
81      public String[] businessKeys() {
82          return new String[]{"commissioner", "responsible", "fromDate", "toDate"};
83      }
84  
85      public void terminate(Date date) {
86          this.toDate = date;
87          save();
88      }
89  
90      @SuppressWarnings("rawtypes")
91      public static <T extends Accountability> List<T> findAccountabilities(Class<T> accountabilityClass, Date date) {
92          return getRepository().createCriteriaQuery(accountabilityClass).le("fromDate", date).gt("toDate", date).list();
93      }
94  
95      @SuppressWarnings("rawtypes")
96      public static List<Accountability> findAccountabilitiesByParty(Party party, Date date) {
97          return getRepository().createNamedQuery("Accountability.findAccountabilitiesByParty")
98                  .addParameter("party", party).addParameter("date", date).list();
99      }
100 
101 }
