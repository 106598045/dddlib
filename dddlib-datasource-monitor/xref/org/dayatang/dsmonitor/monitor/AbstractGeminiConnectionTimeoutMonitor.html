1   package org.dayatang.dsmonitor.monitor;
2   
3   import org.apache.commons.lang3.time.DateFormatUtils;
4   import org.dayatang.dsmonitor.ConnectionMonitor;
5   import org.dayatang.dsmonitor.GeminiConnection;
6   import org.dayatang.utils.Slf4JLogger;
7   
8   import java.sql.SQLException;
9   import java.util.Collections;
10  import java.util.HashSet;
11  import java.util.Set;
12  
13  public abstract class AbstractGeminiConnectionTimeoutMonitor implements ConnectionMonitor {
14  
15  	private final Slf4JLogger LOGGER = Slf4JLogger.getLogger(AbstractGeminiConnectionTimeoutMonitor.class);
16  
17  	private static String DATE_PATTERN = "yyyy-MM-dd HH:mm:ss";
18  
19  	private Set<GeminiConnection> aliveConnections = Collections.synchronizedSet(new HashSet<GeminiConnection>());
20  
21      private int connectionCount;
22  
23      /**
24  	 * 单位：毫秒，默认10000毫秒，即10s
25  	 */
26  	private long timeout = 1000;
27  
28  	public void openConnection(GeminiConnection connection) throws SQLException {
29          LOGGER.debug("开启数据库连接HashCode【{}】，URL=【{}】，创建时间【{}】",
30                  connection.hashCode(), connection.getMetaData().getURL(),
31                  formatTime(connection.getCreationTime()));
32  		aliveConnections.add(connection);
33          connectionCount++;
34  	}
35  
36  	public void closeConnection(GeminiConnection connection) throws SQLException {
37          LOGGER.debug("关闭数据库连接HashCode【{}】，创建时间【{}】，耗时【{}】ms",
38                  connection.hashCode(), formatTime(connection.getCreationTime()),
39                  connection.getSurvivalTime());
40  		aliveConnections.remove(connection);
41  	}
42  
43  	private boolean isTimeout(GeminiConnection connection) {
44          return connection.getSurvivalTime() > timeout;
45  	}
46  
47  	/**
48  	 * 单位：毫秒，默认10000毫秒，即10s
49  	 * 
50  	 * @return 超时时间
51  	 */
52  	public long getTimeout() {
53  		return timeout;
54  	}
55  
56  	/**
57  	 * 设置超时时间，以毫秒为单位
58  	 * 
59  	 * @param timeout 超时时间
60  	 */
61  	public void setTimeout(long timeout) {
62  		this.timeout = timeout;
63  	}
64  
65  	/**
66  	 * 获取活动的超时连接集合
67  	 * 
68  	 * @return 活动的超时连接集合
69  	 */
70  	public Set<GeminiConnection> getAliveTimeoutConnections() {
71  		Set<GeminiConnection> results = new HashSet<GeminiConnection>();
72  		for (GeminiConnection conn : aliveConnections) {
73  			if (isTimeout(conn)) {
74  				results.add(conn);
75  			}
76  		}
77  		return results;
78  	}
79  
80  
81      /**
82       * 获取活动的连接集合
83       *
84       * @return 活动的连接集合
85       */
86      public Set<GeminiConnection> getAliveConnections() {
87          return Collections.unmodifiableSet(aliveConnections);
88      }
89  
90      public int getConnectionCount() {
91          return connectionCount;
92      }
93  
94      private String formatTime(long date) {
95  		return DateFormatUtils.format(date, DATE_PATTERN);
96  	}
97  }
