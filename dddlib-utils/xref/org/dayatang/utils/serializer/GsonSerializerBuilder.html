1   package org.dayatang.utils.serializer;
2   
3   import com.google.gson.*;
4   
5   import java.lang.reflect.Type;
6   import java.util.Arrays;
7   import java.util.List;
8   
9   /**
10   * Created by yyang on 14/11/9.
11   */
12  public class GsonSerializerBuilder {
13      private GsonBuilder builder;
14  
15      public GsonSerializerBuilder() {
16          builder = new GsonBuilder();
17      }
18  
19      public GsonSerializerBuilder(GsonBuilder builder) {
20          this.builder = builder;
21      }
22  
23      public GsonSerializerBuilder serializeNulls() {
24          return new GsonSerializerBuilder(builder.serializeNulls());
25      }
26  
27      public GsonSerializerBuilder excludeFieldsNamed(String... fieldNames) {
28          return new GsonSerializerBuilder(builder
29                  .addSerializationExclusionStrategy(new FieldNameExclusionStrategy(fieldNames)));
30      }
31  
32      public GsonSerializerBuilder excludeFieldsWithModifiers(int... modifiers) {
33          return new GsonSerializerBuilder(builder
34                  .excludeFieldsWithModifiers(modifiers));
35      }
36  
37      public GsonSerializerBuilder addSerializationExclusionStrategy(ExclusionStrategy strategy) {
38          return new GsonSerializerBuilder(builder
39                  .addSerializationExclusionStrategy(strategy));
40      }
41  
42      public GsonSerializerBuilder addDeserializationExclusionStrategy(ExclusionStrategy strategy) {
43          return new GsonSerializerBuilder(builder
44                  .addDeserializationExclusionStrategy(strategy));
45      }
46  
47      public GsonSerializerBuilder registerTypeAdapter(Type type, Object typeAdapter) {
48          return new GsonSerializerBuilder(builder.registerTypeAdapter(type, typeAdapter));
49      }
50  
51      public GsonSerializerBuilder prettyPrinting() {
52          return new GsonSerializerBuilder(builder.setPrettyPrinting());
53      }
54  
55      public GsonSerializerBuilder dateFormat(String pattern) {
56          return new GsonSerializerBuilder(builder.setDateFormat(pattern));
57      }
58  
59      public GsonSerializer build() {
60          return new GsonSerializer(builder.create());
61      }
62  
63      private static class FieldNameExclusionStrategy implements ExclusionStrategy {
64  
65          private List<String> fieldNames;
66  
67          private FieldNameExclusionStrategy(String... fieldNames) {
68              this.fieldNames = Arrays.asList(fieldNames);
69          }
70  
71          @Override
72          public boolean shouldSkipField(FieldAttributes fieldAttributes) {
73              return fieldNames.contains(fieldAttributes.getName());
74          }
75  
76          @Override
77          public boolean shouldSkipClass(Class<?> aClass) {
78              return false;
79          }
80      }
81  
82  }
