71      /**
72       * 计算两个日期（不包括时间）之间相差的周年数
73       *
74       * @param date1
75       * @param date2
76       * @return
77       */
78      public static int getYearDiff(Date date1, Date date2) {
79          if (date1 == null || date2 == null) {
80              throw new InvalidParameterException(
81                      "date1 and date2 cannot be null!");
82          }
83          if (date1.after(date2)) {
84              throw new InvalidParameterException("date1 cannot be after date2!");
85          }
86  
87          Calendar calendar = Calendar.getInstance();
88          calendar.setTime(date1);
89          int year1 = calendar.get(Calendar.YEAR);
90          int month1 = calendar.get(Calendar.MONTH);
91          int day1 = calendar.get(Calendar.DATE);
92  
93          calendar.setTime(date2);
94          int year2 = calendar.get(Calendar.YEAR);
95          int month2 = calendar.get(Calendar.MONTH);
96          int day2 = calendar.get(Calendar.DATE);
97  
98          int result = year2 - year1;
99          if (month2 < month1) {
100             result--;
101         } else if (month2 == month1 && day2 < day1) {
102             result--;
103         }
104         return result;
105     }

107     /**
108      * 计算两个日期（不包括时间）之间相差的整月数
109      *
110      * @param date1
111      * @param date2
112      * @return
113      */
114     public static int getMonthDiff(Date date1, Date date2) {
115         if (date1 == null || date2 == null) {
116             throw new InvalidParameterException(
117                     "date1 and date2 cannot be null!");
118         }
119         if (date1.after(date2)) {
120             throw new InvalidParameterException("date1 cannot be after date2!");
121         }
122 
123         Calendar calendar = Calendar.getInstance();
124         calendar.setTime(date1);
125         int year1 = calendar.get(Calendar.YEAR);
126         int month1 = calendar.get(Calendar.MONTH);
127         int day1 = calendar.get(Calendar.DATE);
128 
129         calendar.setTime(date2);
130         int year2 = calendar.get(Calendar.YEAR);
131         int month2 = calendar.get(Calendar.MONTH);
132         int day2 = calendar.get(Calendar.DATE);
133 
134         int months = 0;
135         if (day2 >= day1) {
136             months = month2 - month1;
137         } else {
138             months = month2 - month1 - 1;
139         }
140         return (year2 - year1) * MONTHS_IN_A_YEAR + months;
141     }

143     /**
144      * 统计两个日期之间包含的天数。包含date1，但不包含date2
145      *
146      * @param date1
147      * @param date2
148      * @return
149      */
150     public static int getDayDiff(Date date1, Date date2) {
151         if (date1 == null || date2 == null) {
152             throw new InvalidParameterException(
153                     "date1 and date2 cannot be null!");
154         }
155         Date startDate = org.apache.commons.lang3.time.DateUtils.truncate(
156                 date1, Calendar.DATE);
157         Date endDate = org.apache.commons.lang3.time.DateUtils.truncate(date2,
158                 Calendar.DATE);
159         if (startDate.after(endDate)) {
160             throw new InvalidParameterException("date1 cannot be after date2!");
161         }
162         long millSecondsInOneDay = HOURS_IN_A_DAY * MINUTES_IN_AN_HOUR
163                 * SECONDS_IN_A_MINUTE * MILLIS_IN_A_SECOND;
164         return (int) ((endDate.getTime() - startDate.getTime()) / millSecondsInOneDay);
165     }

167     /**
168      * 计算time2比time1晚多少分钟，忽略日期部分
169      *
170      * @param time1
171      * @param time2
172      * @return
173      */
174     public static int getMinuteDiffByTime(Date time1, Date time2) {
175         long startMil = 0;
176         long endMil = 0;
177         Calendar calendar = Calendar.getInstance();
178         calendar.setTime(time1);
179         calendar.set(1900, 1, 1);
180         startMil = calendar.getTimeInMillis();
181         calendar.setTime(time2);
182         calendar.set(1900, 1, 1);
183         endMil = calendar.getTimeInMillis();
184         return (int) ((endMil - startMil) / MILLIS_IN_A_SECOND / SECONDS_IN_A_MINUTE);
185     }

332     /**
333      * 计算两个日期之间包含的星期X的天数。
334      *
335      * @param fromDate 起始日期
336      * @param toDate 结束日期
337      * @param dayOfWeek 星期，例如星期三，星期四
338      * @return
339      */
340     public static int getWeekDaysBetween(Date fromDate, Date toDate,
341             int dayOfWeek) {
342         int result = 0;
343         Date firstDate = getFirstWeekdayBetween(fromDate, toDate, dayOfWeek);
344         if (firstDate == null) {
345             return 0;
346         }
347         Calendar aDay = Calendar.getInstance();
348         aDay.setTime(firstDate);
349         while (aDay.getTime().before(toDate)) {
350             result++;
351             aDay.add(Calendar.DATE, DAYS_IN_A_WEEK);
352         }
353         return result;
354     }

356     /**
357      * 获取在两个日期之间的第一个星期X
358      *
359      * @param fromDate 起始日期
360      * @param toDate 结束日期
361      * @param dayOfWeek 星期，例如星期三，星期四
362      * @return
363      */
364     public static Date getFirstWeekdayBetween(Date fromDate, Date toDate,
365             int dayOfWeek) {
366         Calendar aDay = Calendar.getInstance();
367         aDay.setTime(fromDate);
368         while (aDay.getTime().before(toDate)) {
369             if (aDay.get(Calendar.DAY_OF_WEEK) == dayOfWeek) {
370                 return aDay.getTime();
371             }
372             aDay.add(Calendar.DATE, 1);
373         }
374         return null;
375     }
