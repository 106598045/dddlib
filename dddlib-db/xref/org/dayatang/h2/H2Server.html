1   package org.dayatang.h2;
2   
3   import org.dayatang.utils.Slf4JLogger;
4   import org.h2.tools.DeleteDbFiles;
5   import org.h2.tools.Server;
6   
7   import java.io.File;
8   import java.sql.SQLException;
9   
10  public class H2Server {
11  	
12  	private static final Slf4JLogger LOGGER = Slf4JLogger.getLogger(H2Server.class);
13  	private static final String DEFAULT_DB_FILE = "h2-test-db";
14  	private static H2Server instance;
15  	
16  	public static final synchronized H2Server getSingleton() {
17  		if (instance == null) {
18  			instance = new H2Server(DEFAULT_DB_FILE);
19  		}
20  		return instance;
21  	}
22  	
23  	private Server server;
24  
25  	private String dir;
26  
27  	private String dbFile;
28  
29  	public H2Server(String dir, String dbFile) {
30  		this.dir = dir;
31  		this.dbFile = dbFile;
32  	}
33  
34  	public H2Server(String dbFile) {
35  		this(".", dbFile);
36  	}
37  
38  	public synchronized void start() {
39  		if (server != null && server.isRunning(true)) {
40  			return;
41  		}
42  		try {
43  			DeleteDbFiles.execute(dir, dbFile, true);
44  			server = Server.createTcpServer(new String[0]);
45  			server.start();
46  			LOGGER.info("H2 server started! data file is {}", new File(dir, dbFile).getAbsolutePath());
47  		} catch (SQLException e) {
48  			LOGGER.error("Cannot start h2 server database", e);
49  			throw new RuntimeException("Cannot start h2 server database", e);
50  		}
51  	}
52  
53  	public synchronized void shutdown() {
54  		if (server == null) {
55  			return;
56  		}
57  		if (server.isRunning(true)) {
58  			server.stop();
59  			server.shutdown();
60  			DeleteDbFiles.execute(dir, dbFile, true);
61  			LOGGER.info("H2 server shutdowned!");
62  		}
63  		server = null;
64  	}
65  }
