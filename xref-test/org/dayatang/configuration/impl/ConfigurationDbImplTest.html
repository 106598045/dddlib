27  	private static DataSource createDataSource() throws IOException {
28  		Properties properties = new Properties();
29  		properties.load(ConfigurationDbImplTest.class.getResourceAsStream("/jdbc.properties"));
30  		ComboPooledDataSource result = new ComboPooledDataSource();
31  		try {
32  			result.setDriverClass(properties.getProperty("jdbc.driverClassName"));
33  		} catch (PropertyVetoException e) {
34  			throw new RuntimeException("Cannot access JDBC Driver: org.h2.Driver", e);
35  		}
36  		result.setJdbcUrl(properties.getProperty("jdbc.url"));
37  		result.setUser(properties.getProperty("jdbc.username"));
38  		result.setPassword(properties.getProperty("jdbc.password"));
39  		result.setMinPoolSize(10);
40  		result.setMaxPoolSize(300);
41  		result.setInitialPoolSize(10);
42  		result.setAcquireIncrement(5);
43  		result.setMaxStatements(0);
44  		result.setIdleConnectionTestPeriod(60);
45  		result.setAcquireRetryAttempts(30);
46  		result.setBreakAfterAcquireFailure(false);
47  		result.setTestConnectionOnCheckout(false);
48  		return result;
49  	}
