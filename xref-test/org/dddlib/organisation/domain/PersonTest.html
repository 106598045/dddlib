13  public class PersonTest extends AbstractIntegrationTest {
14  
15      private Person person1;
16      private Person person2;
17  
18      @Before
19      public void setUp() throws Exception {
20          super.setUp();
21          person1 = new Person("a", "b");
22          person1.setIdNumber("p1");
23          person1.setIm(ImType.QQ, "349591542");
24          person1.setIm(ImType.MSN, "cnyangyu@hotmail.com");
25          person1.save();
26  
27          person2 = new Person("a", "c");
28          person2.setIdNumber("p2");
29          person2.setIm(ImType.QQ, "666666");
30          person2.setIm(ImType.MSN, "ryyang@hotmail.com");
31          person2.save();
32      }
33  
34      @Test
35      public void testGetIms() {
36          String jpql = "select o from Person o join o.ims i where KEY(i) = :imType and i = :im";
37          EntityRepository repository = InstanceFactory.getInstance(EntityRepository.class);
38          List<Person> persons = repository.createJpqlQuery(jpql)
39                  .addParameter("imType", ImType.QQ).addParameter("im", "666666").list();
40          assertFalse(persons.contains(person1));
41          assertTrue(persons.contains(person2));
42      }
