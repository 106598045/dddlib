1   package org.dayatang.domain.event;
2   
3   import org.dayatang.utils.Assert;
4   
5   import java.util.Date;
6   import java.util.UUID;
7   
8   /**
9    * 领域事件基类，领域事件代表具有业务含义的事件，例如员工调动或者机构调整
10   * Created by yyang on 14-9-12.
11   */
12  public abstract class DomainEvent {
13  
14      private String id = UUID.randomUUID().toString();
15  
16      private Date occurredOn = new Date();
17  
18      private int version = 1;
19  
20      public DomainEvent() {
21          this(new Date(), 1);
22      }
23  
24      /**
25       *
26       * @param occurredOn 发生时间
27       */
28      public DomainEvent(Date occurredOn) {
29          this(occurredOn, 1);
30      }
31  
32      /**
33       *
34       * @param occurredOn 发生时间
35       * @param version 版本
36       */
37      public DomainEvent(Date occurredOn, int version) {
38          Assert.notNull(occurredOn);
39          this.occurredOn = new Date(occurredOn.getTime());
40          this.version = version;
41      }
42  
43      /**
44       * 获得事件ID
45       * @return 事件的ID
46       */
47      public String getId() {
48          return id;
49      }
50  
51      /**
52       * For test only
53       * @param id 要设置的ID
54       */
55      protected void setId(String id) {
56          this.id = id;
57      }
58  
59      /**
60       * 获得事件发生时间
61       * @return 事件发生时间
62       */
63      public Date getOccurredOn() {
64          return occurredOn;
65      }
66  
67      /**
68       * 获得版本
69       * @return 事件的版本
70       */
71      public int getVersion() {
72          return version;
73      }
74  
75      @Override
76      public boolean equals(Object other) {
77          if (this == other) {
78              return true;
79          }
80          if (!(other instanceof DomainEvent)) {
81              return false;
82          }
83          DomainEvent that = (DomainEvent) other;
84          return this.getId().equals(that.getId());
85      }
86  
87      @Override
88      public int hashCode() {
89          return id.hashCode();
90      }
91  }
