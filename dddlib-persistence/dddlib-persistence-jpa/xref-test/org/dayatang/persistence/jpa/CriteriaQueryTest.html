412     private DictionaryCategory createCategory(String name, int sortOrder) {
413         DictionaryCategory category = new DictionaryCategory();
414         category.setName(name);
415         category.setSortOrder(sortOrder);
416         entityManager.persist(category);
417         repository.flush();
418         return category;
419     }
420 
421     private Dictionary createDictionary(String code, String text, DictionaryCategory category, int sortOrder,
422             String parentCode) {
423         Dictionary dictionary = new Dictionary(code, text, category);
424         dictionary.setSortOrder(sortOrder);
425         dictionary.setParentCode(parentCode);
426         entityManager.persist(dictionary);
427         repository.flush();
428         return dictionary;
429     }
