1   package org.dddlib.organisation.domain;
2   
3   import org.dayatang.domain.AbstractEntity;
4   import org.dayatang.utils.Assert;
5   import org.dayatang.utils.DateUtils;
6   
7   import javax.persistence.*;
8   import java.util.Date;
9   
10  @Entity
11  @Inheritance(strategy = InheritanceType.SINGLE_TABLE)
12  @DiscriminatorColumn(name = "CATEGORY", discriminatorType = DiscriminatorType.STRING)
13  public abstract class Party extends AbstractEntity {
14  
15      private static final long serialVersionUID = -6083088250263550905L;
16  
17      // 编码
18      private String sn;
19  
20      // 名称
21      private String name;
22  
23      // 创建日期
24      @Temporal(TemporalType.DATE)
25      private Date createDate = new Date();
26  
27      // 终结日期
28      @Temporal(TemporalType.DATE)
29      private Date terminateDate = DateUtils.MAX_DATE;
30  
31      public Party() {
32      }
33  
34      public Party(String name) {
35          Assert.notBlank(name, "Name is blank!");
36          this.name = name;
37      }
38  
39      public String getSn() {
40          return sn;
41      }
42  
43      public void setSn(String sn) {
44          this.sn = sn;
45      }
46  
47      public String getName() {
48          return name;
49      }
50  
51      public final void setName(String name) {
52          this.name = name;
53      }
54  
55      public Date getCreateDate() {
56          return createDate;
57      }
58  
59      public void setCreateDate(Date createDate) {
60          this.createDate = createDate;
61      }
62  
63      public Date getTerminateDate() {
64          return terminateDate;
65      }
66  
67      public void setTerminateDate(Date terminateDate) {
68          this.terminateDate = terminateDate;
69      }
70  
71      /**
72       * 判断在指定的日期是否“存活”，即参数date处于其生命周期内。
73       *
74       * @param date
75       * @return
76       */
77      public boolean isActive(Date date) {
78          return DateUtils.isSameDay(date, getCreateDate())
79                  || date.after(getCreateDate())
80                  && date.before(getTerminateDate());
81      }
82  
83      /**
84       * 终结当事人，例如撤销机构、撤销岗位、员工离职退休等
85       *
86       * @param date
87       */
88      @SuppressWarnings("rawtypes")
89      public void terminate(Date date) {
90          for (Accountability each : Accountability.findAccountabilitiesByParty(this, date)) {
91              each.terminate(date);
92          }
93          terminateDate = date;
94          save();
95      }
96  
97      @Override
98      public String[] businessKeys() {
99          return new String[]{"sn", "name"};
100     }
101 }
