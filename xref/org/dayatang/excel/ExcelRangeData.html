7   public class ExcelRangeData {
8   	private List<Object[]> data = new ArrayList<Object[]>();
9   	private Version version;
10  	private boolean isDate1904;
11  	
12  	public ExcelRangeData(List<Object[]> data, Version version, boolean isDate1904) {
13  		super();
14  		this.data = data;
15  		this.version = version;
16  		this.isDate1904 = isDate1904;
17  	}
18  	
19  	public int getRowCount() {

21  	}
22  
23  	public int getColumnCount() {
24  		return data.isEmpty() ? 0 : data.get(0).length;
25  	}
26  	
27  	public Double getDouble(int row, int column) {
28  		return ExcelUtils.getDouble(data.get(row)[column]);
29  	}
30  	
31  	public Integer getInt(int row, int column) {
32  		return ExcelUtils.getInt(data.get(row)[column]);
33  	}
34  	
35  	public Long getLong(int row, int column) {
36  		return ExcelUtils.getLong(data.get(row)[column]);
37  	}
38  	
39  	public Boolean getBoolean(int row, int column) {
40  		return ExcelUtils.getBoolean(data.get(row)[column]);
41  	}
42  	
43  	public String getString(int row, int column) {
44  		return ExcelUtils.getString(data.get(row)[column]);
45  	}
46  	
47  	public Date getDate(int row, int column) {
48  		return ExcelUtils.getDate(data.get(row)[column], version, isDate1904);
49  	}
50
