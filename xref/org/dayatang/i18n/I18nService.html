1   package org.dayatang.i18n;
2   
3   import java.util.Locale;
4   
5   public interface I18nService {
6   
7   	/**
8   	 * Try to resolve the message. Return default message if no message was
9   	 * found.
10  	 * 
11  	 * @param code
12  	 *            the code to lookup up, such as 'calculator.noRateSet'. Users
13  	 *            of this class are encouraged to base message names on the
14  	 *            relevant fully qualified class name, thus avoiding conflict
15  	 *            and ensuring maximum clarity.
16  	 * @param args
17  	 *            array of arguments that will be filled in for params within
18  	 *            the message (params look like "{0}", "{1,date}", "{2,time}"
19  	 *            within a message), or <code>null</code> if none.
20  	 * @param defaultMessage
21  	 *            String to return if the lookup fails
22  	 * @param locale
23  	 *            the Locale in which to do the lookup
24  	 * @return the resolved message if the lookup was successful; otherwise the
25  	 *         default message passed as a parameter
26  	 * @see java.text.MessageFormat
27  	 */
28  	String getMessage(String code, Object[] args, String defaultMessage,
29  			Locale locale);
30  
31  	/**
32  	 * Try to resolve the message. Treat as an error if the message can't be
33  	 * found.
34  	 * 
35  	 * @param code
36  	 *            the code to lookup up, such as 'calculator.noRateSet'
37  	 * @param args
38  	 *            Array of arguments that will be filled in for params within
39  	 *            the message (params look like "{0}", "{1,date}", "{2,time}"
40  	 *            within a message), or <code>null</code> if none.
41  	 * @param locale
42  	 *            the Locale in which to do the lookup
43  	 * @return the resolved message
44  	 * @throws NoSuchMessageException
45  	 *             if the message wasn't found
46  	 * @see java.text.MessageFormat
47  	 */
48  	String getMessage(String code, Object[] args, Locale locale)
49  			throws NoSuchMessageException;
50  
51  }
