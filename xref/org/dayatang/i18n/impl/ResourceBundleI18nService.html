17  	/**
18  	 * Cache to hold loaded ResourceBundles. This Map is keyed with the bundle
19  	 * basename, which holds a Map that is keyed with the Locale and in turn
20  	 * holds the ResourceBundle instances. This allows for very efficient hash
21  	 * lookups, significantly faster than the ResourceBundle class's own cache.
22  	 */
23  	private final Map<String, Map<Locale, ResourceBundle>> cachedResourceBundles = new HashMap<String, Map<Locale, ResourceBundle>>();
24  
25  	/**
26  	 * Cache to hold already generated MessageFormats. This Map is keyed with
27  	 * the ResourceBundle, which holds a Map that is keyed with the message
28  	 * code, which in turn holds a Map that is keyed with the Locale and holds
29  	 * the MessageFormat values. This allows for very efficient hash lookups
30  	 * without concatenated keys.
31  	 * 
32  	 * @see #getMessageFormat
33  	 */
34  	private final Map<ResourceBundle, Map<String, Map<Locale, MessageFormat>>> cachedBundleMessageFormats = new HashMap<ResourceBundle, Map<String, Map<Locale, MessageFormat>>>();
35  
36  	/**
37  	 * Set a single basename, following {@link java.util.ResourceBundle}
38  	 * conventions: essentially, a fully-qualified classpath location. If it
39  	 * doesn't contain a package qualifier (such as <code>org.mypackage</code>),
40  	 * it will be resolved from the classpath root.
41  	 * <p>
42  	 * Messages will normally be held in the "/lib" or "/classes" directory of a
43  	 * web application's WAR structure. They can also be held in jar files on
44  	 * the class path.
45  	 * <p>
46  	 * Note that ResourceBundle names are effectively classpath locations: As a
47  	 * consequence, the JDK's standard ResourceBundle treats dots as package
48  	 * separators. This means that "test.theme" is effectively equivalent to
49  	 * "test/theme", just like it is for programmatic
50  	 * <code>java.util.ResourceBundle</code> usage.
51  	 * 
52  	 * @see #setBasenames
53  	 * @see java.util.ResourceBundle#getBundle(String)
54  	 */
55  	public void setBasename(String basename) {
56  		setBasenames(new String[] { basename });
57  	}
58  
59  	/**
60  	 * Set an array of basenames, each following
61  	 * {@link java.util.ResourceBundle} conventions: essentially, a
62  	 * fully-qualified classpath location. If it doesn't contain a package
63  	 * qualifier (such as <code>org.mypackage</code>), it will be resolved from
64  	 * the classpath root.
65  	 * <p>
66  	 * The associated resource bundles will be checked sequentially when
67  	 * resolving a message code. Note that message definitions in a
68  	 * <i>previous</i> resource bundle will override ones in a later bundle, due
69  	 * to the sequential lookup.
70  	 * <p>
71  	 * Note that ResourceBundle names are effectively classpath locations: As a
72  	 * consequence, the JDK's standard ResourceBundle treats dots as package
73  	 * separators. This means that "test.theme" is effectively equivalent to
74  	 * "test/theme", just like it is for programmatic
75  	 * <code>java.util.ResourceBundle</code> usage.
76  	 * 
77  	 * @see #setBasename
78  	 * @see java.util.ResourceBundle#getBundle(String)
79  	 */
80  	public void setBasenames(String[] names) {
81  		if (names == null) {
82  			this.basenames = new String[0];
83  		} else {
84  			this.basenames = Arrays.copyOf(names, names.length);
85  		}
86  		for (int i = 0; i < basenames.length; i++) {
87  			String basename = basenames[i];
88  			if (StringUtils.isEmpty(basename) || StringUtils.isBlank(basename)) {
89  				throw new IllegalArgumentException(
90  						"Basename must not be null or empty");
91  			}
92  			this.basenames[i] = basename.trim();
93  		}
94  	}

126 	/**
127 	 * Resolves the given message code as key in the registered resource
128 	 * bundles, returning the value found in the bundle as-is (without
129 	 * MessageFormat parsing).
130 	 */
131 	@Override
132 	protected String resolveCodeWithoutArguments(String code, Locale locale) {
133 		String result = null;
134 		for (int i = 0; result == null && i < this.basenames.length; i++) {
135 			ResourceBundle bundle = getResourceBundle(this.basenames[i], locale);
136 			if (bundle != null) {
137 				result = getStringOrNull(bundle, code);
138 			}
139 		}
140 		return result;
141 	}

143 	/**
144 	 * Resolves the given message code as key in the registered resource
145 	 * bundles, using a cached MessageFormat instance per message code.
146 	 */
147 	@Override
148 	protected MessageFormat resolveCode(String code, Locale locale) {
149 		MessageFormat messageFormat = null;
150 		for (int i = 0; messageFormat == null && i < this.basenames.length; i++) {
151 			ResourceBundle bundle = getResourceBundle(this.basenames[i], locale);
152 			if (bundle != null) {
153 				messageFormat = getMessageFormat(bundle, code, locale);
154 			}
155 		}
156 		return messageFormat;
157 	}

159 	/**
160 	 * Return a ResourceBundle for the given basename and code, fetching already
161 	 * generated MessageFormats from the cache.
162 	 * 
163 	 * @param basename
164 	 *            the basename of the ResourceBundle
165 	 * @param locale
166 	 *            the Locale to find the ResourceBundle for
167 	 * @return the resulting ResourceBundle, or <code>null</code> if none found
168 	 *         for the given basename and Locale
169 	 */
170 	protected ResourceBundle getResourceBundle(String basename, Locale locale) {
171 		synchronized (this.cachedResourceBundles) {
172 			Map<Locale, ResourceBundle> localeMap = this.cachedResourceBundles
173 					.get(basename);
174 			if (localeMap != null) {
175 				ResourceBundle bundle = localeMap.get(locale);
176 				if (bundle != null) {
177 					return bundle;
178 				}
179 			}
180 			try {
181 				ResourceBundle bundle = doGetBundle(basename, locale);
182 				if (localeMap == null) {
183 					localeMap = new HashMap<Locale, ResourceBundle>();
184 					this.cachedResourceBundles.put(basename, localeMap);
185 				}
186 				localeMap.put(locale, bundle);
187 				return bundle;
188 			} catch (MissingResourceException ex) {
189 				if (logger.isWarnEnabled()) {
190 					logger.warn("ResourceBundle [" + basename
191 							+ "] not found for MessageSource: "
192 							+ ex.getMessage());
193 				}
194 				// Assume bundle not found
195 				// -> do NOT throw the exception to allow for checking parent
196 				// message source.
197 				return null;
198 			}
199 		}
200 	}

203 	 * Obtain the resource bundle for the given basename and Locale.
204 	 * 
205 	 * @param basename
206 	 *            the basename to look for
207 	 * @param locale
208 	 *            the Locale to look for
209 	 * @return the corresponding ResourceBundle
210 	 * @throws MissingResourceException
211 	 *             if no matching bundle could be found
212 	 * @see java.util.ResourceBundle#getBundle(String, java.util.Locale,
213 	 *      ClassLoader)
214 	 * @see #getBundleClassLoader()
215 	 */
216 	protected ResourceBundle doGetBundle(String basename, Locale locale)
217 			throws MissingResourceException {
218 		return ResourceBundle.getBundle(basename, locale,
219 				getBundleClassLoader());
220 	}

222 	/**
223 	 * Return a MessageFormat for the given bundle and code, fetching already
224 	 * generated MessageFormats from the cache.
225 	 * 
226 	 * @param bundle
227 	 *            the ResourceBundle to work on
228 	 * @param code
229 	 *            the message code to retrieve
230 	 * @param locale
231 	 *            the Locale to use to build the MessageFormat
232 	 * @return the resulting MessageFormat, or <code>null</code> if no message
233 	 *         defined for the given code
234 	 * @throws MissingResourceException
235 	 *             if thrown by the ResourceBundle
236 	 */
237 	protected MessageFormat getMessageFormat(ResourceBundle bundle,
238 			String code, Locale locale) throws MissingResourceException {
239 
240 		synchronized (this.cachedBundleMessageFormats) {
241 			Map<String, Map<Locale, MessageFormat>> codeMap = this.cachedBundleMessageFormats
242 					.get(bundle);
243 			Map<Locale, MessageFormat> localeMap = null;
244 			if (codeMap != null) {
245 				localeMap = codeMap.get(code);
246 				if (localeMap != null) {
247 					MessageFormat result = localeMap.get(locale);
248 					if (result != null) {
249 						return result;
250 					}
251 				}
252 			}
253 
254 			String msg = getStringOrNull(bundle, code);
255 			if (msg != null) {
256 				if (codeMap == null) {
257 					codeMap = new HashMap<String, Map<Locale, MessageFormat>>();
258 					this.cachedBundleMessageFormats.put(bundle, codeMap);
259 				}
260 				if (localeMap == null) {
261 					localeMap = new HashMap<Locale, MessageFormat>();
262 					codeMap.put(code, localeMap);
263 				}
264 				MessageFormat result = createMessageFormat(msg, locale);
265 				localeMap.put(locale, result);
266 				return result;
267 			}
268 
269 			return null;
270 		}
271 	}

273 	private String getStringOrNull(ResourceBundle bundle, String key) {
274 		try {
275 			return bundle.getString(key);
276 		} catch (MissingResourceException ex) {
277 			// Assume key not found
278 			// -> do NOT throw the exception to allow for checking parent
279 			// message source.
280 			return null;
281 		}
282 	}

284 	private ClassLoader getDefaultClassLoader() {
285 		ClassLoader cl = null;
286 		try {
287 			cl = Thread.currentThread().getContextClassLoader();
288 		} catch (Exception ex) {
289 			// Cannot access thread context ClassLoader - falling back to system
290 			// class loader...
291 		}
292 		if (cl == null) {
293 			// No thread context class loader -> use class loader of this class.
294 			cl = ResourceBundleI18nService.class.getClassLoader();
295 		}
296 		return cl;
297 	}
