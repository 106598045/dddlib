7   import org.dbunit.operation.DatabaseOperation;
8   import org.hibernate.Session;
9   import org.hibernate.SessionFactory;
10  import org.hibernate.Transaction;
11  import org.junit.BeforeClass;
12  import org.springframework.orm.hibernate4.HibernateTransactionManager;
13  import org.springframework.transaction.TransactionStatus;
14  import org.springframework.transaction.support.DefaultTransactionDefinition;
15  
16  /**
17   * 一个集成测试基类，提供了访问IoC容器（通过Spring实现）和访问数据库（通过DBUnit）的能力。
18   *
19   */
20  public abstract class AbstractIntegratedTestCase extends Dbunit {
21  
22  	protected Session session;
23  
24  	private Transaction tx;
25  
26  	HibernateTransactionManager tm;
27  
28  	TransactionStatus ts;
29  
30  	@BeforeClass
31  	public static void setUpBeforeClass() throws Exception {
32  		InstanceFactory
33  				.setInstanceProvider(new SpringInstanceProvider(springXmlPath()));
34  	}
35  
36  	private static String[] springXmlPath() {
37  		return new String[] { "classpath*:spring/*.xml" };
38  	}
39  
40  	@Override
41  	public final void setUp() throws Exception {
42  		super.setUp();
43  
44  		SessionFactory sessionFactory = InstanceFactory
45  				.getInstance(SessionFactory.class);
46  		tm = InstanceFactory.getInstance(HibernateTransactionManager.class);
47  		if (tm != null) {
48  			ts = tm.getTransaction(new DefaultTransactionDefinition());
49  		}
50  		if (sessionFactory != null) {
51  			session = sessionFactory.openSession();
52  			tx = session.beginTransaction();
53  		}
54  		action4SetUp();
55  	}
56  
57  	@Override
58  	public final void tearDown() throws Exception {
59  		super.tearDown();
60  		if (rollback()) {
61  			if (ts != null && tm != null) {
62  				tm.rollback(ts);
63  			}
64  			if (tx != null && tx.isActive()) {
65  				tx.rollback();
66  			}
67  		} else {
68  			if (ts != null && tm != null) {
69  				tm.commit(ts);
70  			}
71  			if (tx != null && tx.isActive()) {
72  				tx.commit();
73  			}
74  		}
75  		if (session != null && session.isOpen()) {
76  			session.close();
77  		}
78  		session = null;
79  		action4TearDown();
80  	}
81  
82  	/**
83  	 * 设置单元测试方法是否回滚，true=回滚
84  	 * 
85  	 * @return 单元测试方法是否回滚
86  	 */
87  	protected boolean rollback() {
88  		return true;
89  	}
90  
91  	/**
92  	 * 单元测试方法执行前的操作
93  	 */
94  	protected void action4SetUp() {
95  	}
96  
97  	/**
98  	 * 单元测试方法执行后的操作
99  	 */
100 	protected void action4TearDown() {
101 	}
102 
103 	@Override
104 	protected DataSetStrategy getDataSetStrategy() {
105 		return DataSetStrategy.FlatXml;
106 	}
107 
108 	@Override
109 	protected DatabaseOperation setUpOp() {
110 		return DatabaseOperation.CLEAN_INSERT;
111 	}
112 
113 	@Override
114 	protected DatabaseOperation tearDownOp() {
115 		return DatabaseOperation.DELETE_ALL;
116 	}
117 
118 }
