64  	public static List<Observer> findBySubject(Subject subject) {
65  		String queryString = "select o from Observer o where :subjectKey in elements(o.subjectKeys))";
66                  NamedParameters params = NamedParameters.create().add("subjectKey", subject.getSubjectKey());
67  		List<Object> observers = getRepository().createJpqlQuery(queryString)
68                  .addParameter("subjectKey", subject.getSubjectKey()).list();
69  
70  		if (logger.isDebugEnabled()) {
71  			if (observers.isEmpty()) {
72  				logger.debug("没有找到一个观察者：subjectKey为【{}】",
73  						subject.getSubjectKey());
74  			} else {
75  				for (Object observer : observers) {
76  					logger.debug("找到一个观察者：subjectKey为【{}】，observer为【{}】",
77  							subject.getSubjectKey(), observer);
78  				}
79  			}
80  		}
81  		List<Observer> results = new ArrayList<Observer>();
82  		for (Object observer : observers) {
83  			results.add((Observer) observer);
84  		}
85  		return results;
86  	}
