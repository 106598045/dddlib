1   package org.dayatang.ioc.spring.webapp;
2   
3   import org.dayatang.domain.InstanceFactory;
4   import org.dayatang.ioc.spring.factory.SpringInstanceProvider;
5   import org.springframework.web.context.ContextLoaderListener;
6   import org.springframework.web.context.WebApplicationContext;
7   import org.springframework.web.context.support.WebApplicationContextUtils;
8   
9   import javax.servlet.ServletContextEvent;
10  
11  /**
12   * 一个Web监听器，扩展并取代Spring的ContextLoaderListener，将SpringIoC整合到InstanceFactory中。
13   * @author yyang
14   */
15  public class DayatangContextLoaderListener extends ContextLoaderListener {
16  
17          @Override
18  	public void contextInitialized(ServletContextEvent event) {
19  		super.contextInitialized(event);
20  		WebApplicationContext applicationContext = WebApplicationContextUtils.getRequiredWebApplicationContext(event.getServletContext());
21  		SpringInstanceProvider springProvider = new SpringInstanceProvider(applicationContext);
22  		InstanceFactory.setInstanceProvider(springProvider);
23  	}
24  
25  }
