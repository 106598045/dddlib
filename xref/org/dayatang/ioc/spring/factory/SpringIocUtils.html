19  		SpringInstanceProvider result = providerHolder.get();
20          if (result != null) {
21              return result;
22          }
23          synchronized (SpringIocUtils.class) {
24              result = new SpringInstanceProvider(annotatedClasses);
25              providerHolder.set(result);

27          }
28  	}
