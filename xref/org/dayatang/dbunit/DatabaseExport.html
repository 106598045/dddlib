24  public class DatabaseExport {
25  
26  	private DatabaseExport() {
27  		super();
28  	}
29  
30  	public static IDatabaseConnection createConnection() throws Exception {
31  		// database connection
32  		Class.forName(PropertiesUtil.JDBC_DRIVER);
33  		Connection jdbcConnection = DriverManager.getConnection(
34  				PropertiesUtil.JDBC_URL, PropertiesUtil.JDBC_USERNAME,
35  				PropertiesUtil.JDBC_PASSWD);
36  		return new DatabaseConnection(jdbcConnection);
37  	}
38  
39  	public static void exportDataSet(IDataSet dataSet, String fileName)
40  			throws DataSetException, FileNotFoundException, IOException {
41  		FlatXmlDataSet.write(dataSet, new FileOutputStream(fileName));
42  	}
43  
44  }
