26  /**
27   * DBUnit实用工具类。用于的在数据库和XML数据文件之间的相互输入输出。
28   * XML数据文件只支持FlatXml格式，则数据库中的每行数据表示为一个XML元素，
29   * 每个列成为该元素的一个属性。
30   * @author yyang (<a href="mailto:gdyangyu@gmail.com">gdyangyu@gmail.com</a>)
31   *
32   */
33  public class DbUnitUtils {
34  	
35  	private DataSource dataSource;
36  	
37  	/**
38  	 * 从类路径属性文件中读入JDBC连接信息
39  	 * @param resourceFile
40  	 * @return
41  	 */
42  	public static DbUnitUtils configFromClasspath(String resourceFile) {
43  		Properties jdbcProperties = PropertiesReader.readPropertiesFromClasspath(resourceFile);
44  		return new DbUnitUtils(createDataSource(jdbcProperties));
45  	}
46  
47  	/**
48  	 * 从磁盘文件中读入JDBC连接信息
49  	 * @param configFile
50  	 * @return
51  	 */
52  	public static DbUnitUtils configFromFile(String configFile) {
53  		Properties jdbcProperties = PropertiesReader.readPropertiesFromFile(configFile);
54  		return new DbUnitUtils(createDataSource(jdbcProperties));
55  	}
56  
57  	public DbUnitUtils(DataSource dataSource) {
58  		this.dataSource = dataSource;
59  	}
60  
61  	/**
62  	 * 从XML数据文件中读入数据集，写入到数据库。数据库表中的原有数据将被清除。
63  	 * @param flatXmlDataFile XML数据文件
64  	 */
65  	public void importDataFromClasspath(final String flatXmlDataFile) {
66  		new DbUnitTemplate(dataSource).execute(new DbUnitCallback() {
67  			@Override
68  			public void doInDbUnit(IDatabaseConnection connection) throws Exception {
69  				DatabaseOperation.CLEAN_INSERT.execute(connection, getDatasetFromFile(flatXmlDataFile));
70  			}
71  		});
72  	}
73  
74  	/**
75  	 * 将数据库中的数据导出到XML文件中。
76  	 * @param dir 文件存放目录
77  	 * @param fileName 生成的XML数据文件名。
78  	 */
79  	public void exportData(final String dir, final String fileName) {
80  		new DbUnitTemplate(dataSource).execute(new DbUnitCallback() {
81  			@Override
82  			public void doInDbUnit(IDatabaseConnection connection) throws Exception {
83  				IDataSet dataSet = new FilteredDataSet(new DatabaseSequenceFilter(connection), 
84  						connection.createDataSet());
85  				File parent = new File(dir);
86  				if (!parent.exists()) {
87  					parent.mkdirs();
88  				}
89  				OutputStream out = new FileOutputStream(new File(dir, fileName));
90  				FlatXmlDataSet.write(dataSet, out);
91  			}
92  		});
93  	}
94  
95  	/**
96  	 * 读取数据库表结构，生成XML数据文件的DTD文件
97  	 * @param dir 文件存放目录
98  	 * @param fileName 生成的XML数据文件名。
99  	 */
100 	public void exportDtd(final String dir, final String fileName) {
101 		new DbUnitTemplate(dataSource).execute(new DbUnitCallback() {
102 			@Override
103 			public void doInDbUnit(IDatabaseConnection connection) throws Exception {
104 				IDataSet dataSet = new FilteredDataSet(new DatabaseSequenceFilter(connection), 
105 						connection.createDataSet());
106 				File parent = new File(dir);
107 				if (!parent.exists()) {
108 					parent.mkdirs();
109 				}
110 		        Writer out = new OutputStreamWriter(new FileOutputStream(new File(dir, fileName)));
111 		        FlatDtdDataSet.write(dataSet, out);
112 			}
113 		});
114 	}
115 	
116 	/**
117 	 * 从XML数据文件中读入数据集，写入到数据库。数据库表中的原有数据将被清除。
118 	 * @param flatXmlDataFile XML数据文件
119 	 */
120 	public void refreshData(final String flatXmlDataFile) {
121 		new DbUnitTemplate(dataSource).execute(new DbUnitCallback() {
122 			@Override
123 			public void doInDbUnit(IDatabaseConnection connection) throws Exception {
124 				InputStream in = getClass().getResourceAsStream(flatXmlDataFile);
125 				IDataSetProducer producer = new FlatXmlProducer(new InputSource(in), false);
126 				IDataSet dataSet = new StreamingDataSet(producer);
127 				DatabaseOperation.REFRESH.execute(connection, dataSet);
128 			}
129 		});
130 	}
131 	
132 	private IDataSet getDatasetFromFile(String flatXmlDataFile) {
133 		return getDatasetFromInputStream(getClass().getResourceAsStream(flatXmlDataFile));
134 	}
135 
136 	private IDataSet getDatasetFromInputStream(InputStream in) {
137 		IDataSetProducer producer = new FlatXmlProducer(new InputSource(in), false);
138 		try {
139 			return new CachedDataSet(producer);
140 		} catch (DataSetException e) {
141 			throw new RuntimeException("Cannot get dataset.", e);
142 		}
143 	}
144 
145 	private static DataSource createDataSource(Properties jdbcProperties) {
146 		ComboPooledDataSource dataSource = new ComboPooledDataSource();
147 		try {
148 			dataSource.setDriverClass(jdbcProperties.getProperty(JdbcConstants.JDBC_DRIVER));
149 		} catch (PropertyVetoException e) {
150 			throw new RuntimeException("Cannot create C3P0 data source", e);
151 		}
152 		dataSource.setJdbcUrl(jdbcProperties.getProperty(JdbcConstants.JDBC_URL));
153 		dataSource.setUser(jdbcProperties.getProperty(JdbcConstants.JDBC_USERNAME));
154 		dataSource.setPassword(jdbcProperties.getProperty(JdbcConstants.JDBC_PASSWORD));
155 		return dataSource;
156 	}
