1   package org.dayatang.datasource4saas.dsregistry;
2   
3   import org.apache.commons.lang3.StringUtils;
4   import org.dayatang.utils.Slf4JLogger;
5   
6   import javax.naming.Context;
7   import javax.naming.InitialContext;
8   import javax.naming.NamingException;
9   import javax.sql.DataSource;
10  
11  /**
12   * 数据源注册表实现，将租户数据源映射到JNDI
13   * 
14   * @author yyang (<a href="mailto:gdyangyu@gmail.com">gdyangyu@gmail.com</a>)
15   * 
16   */
17  public class JndiMappingDataSourceRegistry extends MapBasedDataSourceRegistry {
18  
19  	private static final Slf4JLogger LOGGER = Slf4JLogger.getLogger(JndiMappingDataSourceRegistry.class);
20  	private Context context;
21  	private String jndiPrefix;
22  	private String jndiSuffix;
23  
24  	private Context getContext() {
25  		if (context == null) {
26  			try {
27  				context = new InitialContext();
28  			} catch (NamingException e) {
29  				throw new DataSourceRegistryException("Cannot initiate JNDI environment!", e);
30  			}
31  		}
32  		return context;
33  	}
34  
35  	public void setContext(Context context) {
36  		this.context = context;
37  	}
38  
39  	public void setJndiPrefix(String jndiPrefix) {
40  		this.jndiPrefix = jndiPrefix;
41  	}
42  
43  	public void setJndiSuffix(String jndiSuffix) {
44  		this.jndiSuffix = jndiSuffix;
45  	}
46  
47  	/**
48  	 * 根据租户名称，拼装上相应的前缀和后缀作为数据源JNDI名称，从应用服务器上获取已部署的数据源。
49  	 */
50  	@Override
51  	public DataSource findOrCreateDataSourceForTenant(String tenant) {
52  		DataSource result = null;
53  		String dataSourceJndi = StringUtils.defaultString(jndiPrefix) + tenant + StringUtils.defaultString(jndiSuffix);
54  		try {
55  			result = (DataSource) getContext().lookup(dataSourceJndi);
56  		} catch (NamingException e) {
57  			String message = "Lookup jndi: " + dataSourceJndi + " failed!";
58  			LOGGER.error(message, e);
59  			throw new DataSourceRegistryException(message, e);
60  		}
61  		if (result == null) {
62  			throw new DataSourceRegistryException("There's no data source prepared for tenant " + tenant);
63  		}
64  		LOGGER.info("Found JNDI " + dataSourceJndi + " for tenant {}", tenant);
65  		return result;
66  	}
67  }
