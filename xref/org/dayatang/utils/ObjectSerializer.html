1   package org.dayatang.utils;

3   /**
4    * 对象序列化器，将Java对象序列化为字符串形式，或者相反，从字符串反序列化为对象。
5    * Created by yyang on 14-9-16.
6    */
7   public interface ObjectSerializer {
8   
9       /**
10       * 将对象序列化为字符串
11       * @param anObject 要序列化的对象
12       * @return 对象的序列化形式
13       */
14      String serialize(Object anObject);
15  
16      /**
17       * 将字符串反序列化为对象
18       * @param serializedString 对象的字符串序列化形式
19       * @param objectClass 对象的类
20       * @param <T> 对象的类型
21       * @return 一个对象实例
22       */
23      <T> T deserialize(String serializedString, Class<T> objectClass);
24  }
