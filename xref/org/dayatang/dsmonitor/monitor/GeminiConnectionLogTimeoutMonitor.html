1   package org.dayatang.dsmonitor.monitor;
2   
3   import org.apache.commons.lang3.time.DateFormatUtils;
4   import org.dayatang.dsmonitor.GeminiConnection;
5   import org.dayatang.utils.Slf4JLogger;
6   
7   import java.sql.SQLException;
8   import java.util.Set;
9   
10  public class GeminiConnectionLogTimeoutMonitor extends
11  		AbstractGeminiConnectionTimeoutMonitor {
12  
13  	private final Slf4JLogger LOGGER = Slf4JLogger
14  			.getLogger(GeminiConnectionLogTimeoutMonitor.class);
15  
16  	private static String DATE_PATTERN = "yyyy-MM-dd HH:mm:ss";
17  
18  	@Override
19  	public void monitor() {
20  		logConnections(getAliveTimeoutConnections(), "活动的超时数据库连接");
21  	}
22  
23  	private void logConnections(Set<GeminiConnection> connections, String title) {
24  		if (connections.isEmpty()) {
25              LOGGER.info("没有【{}】，超时时间为：【{}】ms", title, getTimeout());
26  			return;
27  		}
28          LOGGER.info("=======================================================================================");
29          LOGGER.info("=======================================================================================");
30          LOGGER.info("=============                                                             =============");
31          LOGGER.info("=============            以下是{}，超时时间为：【{}】ms         =============", title, getTimeout());
32          LOGGER.info("=============                                                             =============");
33          LOGGER.info("=======================================================================================");
34          LOGGER.info("=======================================================================================");
35  
36  		for (GeminiConnection conn : connections) {
37  			logConnection(conn);
38  		}
39  	}
40  
41  	private void logConnection(GeminiConnection connection) {
42  		try {
43  			if (connection.isClosed()) {
44                  LOGGER.info("数据库连接HashCode【{}】，创建时间【{}】，耗时【{}】毫秒", connection.hashCode(),
45                          formatTime(connection.getCreationTime()), connection.getSurvivalTime());
46  			} else {
47                  LOGGER.info("数据库连接HashCode【{}】，URL=【{}】，创建时间【{}】，耗时【{}】毫秒",
48                          connection.hashCode(), connection.getMetaData().getURL(),
49                          formatTime(connection.getCreationTime()), connection.getSurvivalTime());
50  			}
51  
52              LOGGER.info("调用堆栈为：");
53  			for (StackTraceElement ste : connection.getStackTraceElements()) {
54                  LOGGER.info("     " + ste);
55  			}
56  		} catch (SQLException ex) {
57  			ex.printStackTrace();
58  		}
59  	}
60  
61  	private String formatTime(long date) {
62  		return DateFormatUtils.format(date, DATE_PATTERN);
63  	}
64  }
