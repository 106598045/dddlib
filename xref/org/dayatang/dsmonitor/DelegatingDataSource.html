1   package org.dayatang.dsmonitor;
2   
3   import org.dayatang.utils.Assert;
4   import org.dayatang.utils.BeanUtils;
5   
6   import javax.sql.DataSource;
7   import java.io.PrintWriter;
8   import java.sql.Connection;
9   import java.sql.SQLException;
10  import java.sql.SQLFeatureNotSupportedException;
11  import java.util.logging.Logger;
12  
13  /**
14   * 来自spring
15   * 
16   * @author chencao
17   * 
18   */
19  public class DelegatingDataSource implements DataSource {
20  
21  	private DataSource targetDataSource;
22  
23  	/**
24  	 * Create a new DelegatingDataSource.
25  	 * 
26  	 * @param targetDataSource
27  	 *            the target DataSource
28  	 */
29  	public DelegatingDataSource(DataSource targetDataSource) {
30  		Assert.notNull(targetDataSource, "'targetDataSource' must not be null");
31  		this.targetDataSource = targetDataSource;
32  	}
33  
34  	/**
35  	 * Return the target DataSource that this DataSource should delegate to.
36  	 */
37  	public DataSource getTargetDataSource() {
38  		return this.targetDataSource;
39  	}
40  
41  	public Connection getConnection() throws SQLException {
42  		return getTargetDataSource().getConnection();
43  	}
44  
45  	public Connection getConnection(String username, String password)
46  			throws SQLException {
47  		return getTargetDataSource().getConnection(username, password);
48  	}
49  
50  	public PrintWriter getLogWriter() throws SQLException {
51  		return getTargetDataSource().getLogWriter();
52  	}
53  
54  	public void setLogWriter(PrintWriter out) throws SQLException {
55  		getTargetDataSource().setLogWriter(out);
56  	}
57  
58  	public int getLoginTimeout() throws SQLException {
59  		return getTargetDataSource().getLoginTimeout();
60  	}
61  
62  	public void setLoginTimeout(int seconds) throws SQLException {
63  		getTargetDataSource().setLoginTimeout(seconds);
64  	}
65  
66  	// ---------------------------------------------------------------------
67  	// Implementation of JDBC 4.0's Wrapper interface
68  	// ---------------------------------------------------------------------
69  
70  	public <T> T unwrap(Class<T> iface) throws SQLException {
71  		return getTargetDataSource().unwrap(iface);
72  	}
73  
74  	public boolean isWrapperFor(Class<?> iface) throws SQLException {
75  		return getTargetDataSource().isWrapperFor(iface);
76  	}
77  
78  	//For JDK 7 compatability
79  	public Logger getParentLogger() throws SQLFeatureNotSupportedException {
80          BeanUtils beanUtils = new BeanUtils(targetDataSource);
81  		return (Logger) beanUtils.getPropValue("parentLogger");
82  	}
83  
84  }
