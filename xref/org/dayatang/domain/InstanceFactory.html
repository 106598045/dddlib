6   import javax.inject.Named;
7   import java.lang.annotation.Annotation;
8   import java.util.*;

10  /**
11   * <p>
12   * 实例工厂类，充当IoC容器的门面，通过它可以获得部署在IoC容器中的Bean的实例。 InstanceFactory向客户代码隐藏了IoC
13   * 工厂的具体实现。在后台，它通过InstanceProvider策略接口，允许选择不同的IoC工厂，例如Spring， Google Guice和
14   * TapestryIoC等等。
15   * <p>
16   * IoC工厂应该在应用程序启动时装配好，也就是把初始化好的InstanceProvider实现类提供给InstanceFactory。对于web应用
17   * 来说，最佳的初始化方式是创建一个Servlet过滤器或监听器，并部署到web.xml里面；对普通java应用程序来说，最佳的初始化
18   * 位置是在main()函数里面；对于单元测试，最佳的初始化位置是@BeforeClass或@Before标注的方法内部。<br>
19   * <p>
20   * InstanceFactor顺序通过三种途径获取Bean实例。（1）如果已经给InstanceFactory设置了InstanceProvider，那么就通过后者
21   * 查找Bean；（2）如果没有设置InstanceProvider，或者通过InstanceProvider无法找到Bean，就通过JDK6的ServiceLoader查找（通
22   * 过在类路径或jar中的/META-INF/services/a.b.c.Abc文件中设定内容为x.y.z.Xyz，就表明类型a.b.c.Abc将通过类x.y.z.Xyz
23   * 的实例提供）；（3）如果仍然没找到Bean实例，那么将返回那些通过bind()方法设置的Bean实例。（4）如果最终仍然找不到，就抛出
24   * IocInstanceNotFoundException异常。
25   *
26   * @author yyang (<a href="mailto:gdyangyu@gmail.com">gdyangyu@gmail.com</a>)
27   */
28  public class InstanceFactory {
29  
30      private static final Logger LOGGER = LoggerFactory.getLogger(InstanceFactory.class);
31  
32      /**
33       * 以下部分仅用于提供代码测试功能，产品代码不要用
34       */
35      private static final Map<Object, Object> instances = new HashMap<Object, Object>();
36  
37  
38      //实例提供者，代表真正的IoC容器
39      private static InstanceProvider instanceProvider;
40  
41      private static InstanceLocatorFactory instanceLocatorFactory = ServiceLoader.load(InstanceLocatorFactory.class).iterator().next();
42  
43      private static List<InstanceLocator> instanceLocators = new ArrayList<InstanceLocator>();
44  
45      static {
46          instanceLocators.add(instanceLocatorFactory.createByServiceLoader());
47          instanceLocators.add(instanceLocatorFactory.create(instances));
48      }
49  
50      private InstanceFactory() {
51      }
52  
53      /**
54       * 设置实例提供者。
55       *
56       * @param provider 一个实例提供者的实例。
57       */
58      public static void setInstanceProvider(InstanceProvider provider) {
59          instanceProvider = provider;
60          if (instanceProvider == null) {
61              return;
62          }
63          instanceLocators.add(0, instanceLocatorFactory.create(instanceProvider));
64      }
65  
66      /**
67       * 根据类型获取对象实例。返回的对象实例所属的类是T或它的实现类或子类。如果找不到该类型的实例则抛出异常。
68       *
69       * @param <T> 对象的类型
70       * @param beanType 对象所属的类型
71       * @return 类型为T的对象实例
72       */
73      @SuppressWarnings("unchecked")
74      public static <T> T getInstance(Class<T> beanType) {
75          for (InstanceLocator locator : instanceLocators) {
76              T result = locator.getInstance(beanType);
77              if (result != null) {
78                  return result;
79              }
80          }
81          throw new IocInstanceNotFoundException("There's not bean of type '" + beanType + "' exists in IoC container!");
82      }
83  
84      /**
85       * 根据类型和名称获取对象实例。返回的对象实例所属的类是T或它的实现类或子类。不同的IoC容器用不同的方式解释beanName。
86       * 具体的解释方式请参见各种InstanceProvider实现类的Javadoc。 如果找不到该类型的实例则抛出异常。
87       *
88       * @param <T> 类型参数
89       * @param beanName bean的名称
90       * @param beanType 实例的类型
91       * @return 指定类型的实例。
92       */
93      @SuppressWarnings("unchecked")
94      public static <T> T getInstance(Class<T> beanType, String beanName) {
95          for (InstanceLocator locator : instanceLocators) {
96              T result = locator.getInstance(beanType, beanName);
97              if (result != null) {
98                  return result;
99              }
100         }
101         throw new IocInstanceNotFoundException("There's not bean of type '" + beanType + "' exists in IoC container!");
102     }
103 
104     /**
105      * 根据类型和Annotation获取对象实例。返回的对象实例所属的类是T或它的实现类或子类。不同的IoC容器用不同的方式解释annotation。
106      * 具体的解释方式请参见各种InstanceProvider实现类的Javadoc。 如果找不到该类型的实例则抛出异常。
107      *
108      * @param <T> 类型参数
109      * @param beanType 实例的类型
110      * @param annotationType 实现类的annotation类型
111      * @return 指定类型的实例。
112      */
113     public static <T> T getInstance(Class<T> beanType, Class<? extends Annotation> annotationType) {
114         for (InstanceLocator locator : instanceLocators) {
115             T result = locator.getInstance(beanType, annotationType);
116             if (result != null) {
117                 return result;
118             }
119         }
120         throw new IocInstanceNotFoundException("There's not bean '"
121                 + annotationType + "' of type '" + beanType + "' exists in IoC container!");
122     }
123 
124     /**
125      * 将服务绑定到具体实例
126      *
127      * @param <T> Bean实例的类型
128      * @param serviceInterface 注册类型
129      * @param serviceImplementation 对象实例
130      */
131     public static <T> void bind(Class<T> serviceInterface, T serviceImplementation) {
132         instances.put(serviceInterface, serviceImplementation);
133     }
134 
135     /**
136      * 将服务绑定到具体实例并指定名字
137      *
138      * @param <T> Bean实例的类型
139      * @param serviceInterface 注册类型
140      * @param serviceImplementation 对象实例
141      * @param beanName 实例名称
142      */
143     public static <T> void bind(Class<T> serviceInterface, T serviceImplementation, String beanName) {
144         instances.put(toName(serviceInterface, beanName), serviceImplementation);
145     }
146 
147     /**
148      * 删除缓存的bean实例
149      */
150     public static void clear() {
151         instances.clear();
152     }
153 
154     /**
155      * 将服务绑定到具体实例并指定关联的Annotation
156      *
157      * @param <T> Bean实例的类型
158      * @param serviceInterface 注册类型
159      * @param serviceImplementation 对象实例
160      * @param annotationType 标注类型
161      */
162     public static <T> void bind(Class<T> serviceInterface, T serviceImplementation, Class<? extends Annotation> annotationType) {
163         instances.put(toName(serviceInterface, annotationType), serviceImplementation);
164     }
165 
166     private static String toName(Class<?> beanType, String beanName) {
167         return beanType.getName() + ":" + beanName;
168     }
169 
170     private static String toName(Class<?> beanType, Class<? extends Annotation> annotationType) {
171         return beanType.getName() + ":" + annotationType.getName();
172     }
173 }
