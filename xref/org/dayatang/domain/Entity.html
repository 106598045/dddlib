1   package org.dayatang.domain;
2   
3   import java.io.Serializable;
4   
5   /**
6    * 领域实体接口。所有实体类都要直接或间接实现这个接口。它主要起标记作用，以便于统一处理系统中的实体等。
7    * @author yyang (<a href="mailto:gdyangyu@gmail.com">gdyangyu@gmail.com</a>)
8    * 
9    */
10  public interface Entity extends Serializable {
11  
12  	/**
13  	 * 取得实体的Id。实体类的每个实例都必须有个唯一Id以标识自身。
14  	 * 实体Id必须是可序列化的。
15  	 * @return 实体实例的 Id.
16  	 */
17  	Serializable getId();
18  	
19  	/**
20  	 * 是否在数据库中已经存在
21  	 * @return 如果该实体以存在于数据库中，返回true，否则返回false
22  	 */
23  	boolean existed();
24  	
25  	/**
26  	 * 是否在数据库中不存在
27       * @return 如果该实体以存在于数据库中，返回false，否则返回true
28  	 */
29  	boolean notExisted();
30  }
