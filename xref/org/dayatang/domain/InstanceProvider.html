1   package org.dayatang.domain;
2   
3   
4   import java.lang.annotation.Annotation;
5   
6   /**
7    * 实例提供者接口，其实现类以适配器的方式将Bean查找的任务委托给真正的IoC容器，如SpringIoC或Google Guice。
8    * @author yyang (<a href="mailto:gdyangyu@gmail.com">gdyangyu@gmail.com</a>)
9    *
10   */
11  public interface InstanceProvider {
12  
13  	/**
14       * 根据类型获取对象实例。返回的对象实例所属的类是T或它的实现类或子类。如果找不到该类型的实例则抛出异常。
15  	 * @param <T> 类型参数
16  	 * @param beanType 实例的类型
17  	 * @return 指定类型的实例。
18  	 */
19  	<T> T getInstance(Class<T> beanType);
20  
21  	/**
22       * 根据类型和名称获取对象实例。返回的对象实例所属的类是T或它的实现类或子类。不同的IoC容器用不同的方式解释beanName。
23       * 具体的解释方式请参见各种InstanceProvider实现类的Javadoc。
24       * 如果找不到该类型的实例则抛出异常。
25  	 * @param <T> 类型参数
26  	 * @param beanName 实现类在容器中配置的名字
27  	 * @param beanType 实例的类型
28  	 * @return 指定类型的实例。
29  	 */
30  	<T> T getInstance(Class<T> beanType, String beanName);
31  
32      /**
33       * 根据类型和Annotation获取对象实例。返回的对象实例所属的类是T或它的实现类或子类。不同的IoC容器用不同的方式解释annotation。
34       * 具体的解释方式请参见各种InstanceProvider实现类的Javadoc。
35       * 如果找不到该类型的实例则抛出异常。
36       * @param <T> 类型参数
37       * @param beanType 实例的类型
38       * @param annotationType 实现类的annotation类型
39       * @return 指定类型的实例。
40       */
41      <T> T getInstance(Class<T> beanType, Class<? extends Annotation> annotationType);
42  }
