4   package org.dayatang.domain;
5   
6   import java.util.Map;
7   import javax.persistence.*;
8   import org.apache.commons.lang3.builder.EqualsBuilder;
9   import org.apache.commons.lang3.builder.HashCodeBuilder;
10  import org.dayatang.utils.BeanUtils;
11  
12  /**
13   * 抽象实体类，可作为所有领域实体的基类。
14   *
15   * @author yang
16   *
17   */
18  @MappedSuperclass
19  public abstract class BaseEntity implements Entity {
20  
21      private static final long serialVersionUID = 8882145540383345037L;
22  
23      /**
24       * 判断该实体是否已经存在于数据库中。
25       * @return 如果数据库中已经存在拥有该id的实体则返回true，否则返回false。
26       */
27      @Override
28      public boolean existed() {
29          Object id = getId();
30          if (id == null) {
31              return false;
32          }
33          if (id instanceof Number && ((Number)id).intValue() == 0) {
34              return false;
35          }
36          return getRepository().exists(getClass(), getId());
37      }
38  
39      /**
40       * 判断该实体是否不存在于数据库中。
41       * @return 如果数据库中已经存在拥有该id的实体则返回false，否则返回true。
42       */
43      @Override
44      public boolean notExisted() {
45          return !existed();
46      }
47  
48      private static EntityRepository repository;
49  
50      /**
51       * 获取仓储对象实例。如果尚未拥有仓储实例则通过InstanceFactory向IoC容器获取一个。
52       * @return 仓储对象实例
53       */
54      public static EntityRepository getRepository() {
55          if (repository == null) {
56              repository = InstanceFactory.getInstance(EntityRepository.class);
57          }
58          return repository;
59      }
60  
61      /**
62       * 设置仓储实例。该方法主要用于单元测试。产品系统中通常是通过IoC容器获取仓储实例。
63       * @param repository 要设置的仓储对象实例
64       */
65      public static void setRepository(EntityRepository repository) {
66          BaseEntity.repository = repository;
67      }
68      
69      /**
70       * 获取业务主键。业务主键是判断相同类型的两个实体在业务上的等价性的依据。如果相同类型的两个
71       * 实体的业务主键相同，则认为两个实体是相同的，代表同一个实体。
72       * <p>业务主键由实体的一个或多个属性组成。
73       * @return 组成业务主键的属性的数组。
74       */
75      public String[] businessKeys() {
76          return new String[] {};
77      }
78  
79      /**
80       * 依据业务主键获取哈希值。用于判定两个实体是否等价。
81       * 等价的两个实体的hashCode相同，不等价的两个实体hashCode不同。
82       * @return 实体的哈希值
83       */
84      @Override
85      public int hashCode() {
86          HashCodeBuilder builder = new HashCodeBuilder(13, 37);
87          Map<String, Object> propValues = new BeanUtils(this).getPropValues();
88          
89          for (String businessKey : businessKeys()) {
90              builder = builder.append(propValues.get(businessKey));
91          }
92          return builder.toHashCode();
93      }
94  
95      /**
96       * 依据业务主键判断两个实体是否等价。
97       * @param other 另一个实体
98       * @return 如果本实体和other等价则返回true,否则返回false
99       */
100     @Override
101     public boolean equals(Object other) {
102         if (this == other) {
103             return true;
104         }
105         if (other == null) {
106             return false;
107         }
108         if (businessKeys() == null || businessKeys().length == 0) {
109             return false;
110         }
111         if (!(this.getClass().isAssignableFrom(other.getClass()))) {
112             return false;
113         }
114         Map<String, Object> thisPropValues = new BeanUtils(this).getPropValuesExclude(Transient.class);
115         Map<String, Object> otherPropValues = new BeanUtils(other).getPropValuesExclude(Transient.class);
116         EqualsBuilder builder = new EqualsBuilder();
117         for (String businessKey : businessKeys()) {
118             builder.append(thisPropValues.get(businessKey), otherPropValues.get(businessKey));
119         }
120         return builder.isEquals();
121     }
122 }
