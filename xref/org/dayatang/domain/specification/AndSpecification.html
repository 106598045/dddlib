1   package org.dayatang.domain.specification;

3   import org.dayatang.utils.Assert;
4   
5   /**
6    * AND规范，用于创建一个新规范，作为其他两个规范的"与（AND）"操作的结果。
7    *
8    * @param <T> 类型参数，表示规范所应用到的目标对象类型
9    */
10  public class AndSpecification<T> extends AbstractSpecification<T> {
11  
12      private final Specification<T> spec1;
13      private final Specification<T> spec2;
14  
15      /**
16       * 基于其他两个规范创建它们的AND规范。
17       *
18       * @param spec1 第一个规范.
19       * @param spec2 第二个规范.
20       */
21      public AndSpecification(final Specification<T> spec1, final Specification<T> spec2) {
22          Assert.notNull(spec1, "Specification " + spec1 + " is null!");
23          Assert.notNull(spec2, "Specification " + spec2 + " is null!");
24          this.spec1 = spec1;
25          this.spec2 = spec2;
26      }
27  
28      /**
29       * {@inheritDoc}
30       *
31       * @param t 要用来判断是否满足本AND规范的对象。
32       * @return 如果对象t同时满足规范1和规范2两个规范则返回true,否则返回false。
33       */
34      @Override
35      public boolean isSatisfiedBy(final T t) {
36          return spec1.isSatisfiedBy(t) && spec2.isSatisfiedBy(t);
37      }
38  }
