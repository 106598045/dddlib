1   package org.dayatang.domain.specification;
2   
3   /**
4    * 业务规范接口，对应于DDD一书中的“规范”概念，用于判断业务规则是否得到满足。
5    * @param <T> 类型参数，表示规范所应用到的目标对象类型
6    */
7   public interface Specification<T> {
8   
9     /**
10     * 检查 {@code t} 是否由本规范满足.
11     *
12     * @param t 要测试的对象.
13     * @return 如果{@code t}满足此规范则返回{@code true}。
14     */
15    boolean isSatisfiedBy(T t);
16  
17    /**
18     * 创建一个新规范，作为本规范{@code this}和另一个规范{@code specification}的“与（AND）”操作的结果
19     * @param specification 另一个规范。
20     * @return 一个新规范。
21     */
22    Specification<T> and(Specification<T> specification);
23  
24    /**
25     * 创建一个新规范，作为本规范{@code this}和另一个规范{@code specification}的“或（OR）”操作的结果
26     * @param specification 另一个规范。
27     * @return 一个新规范。
28     */
29    Specification<T> or(Specification<T> specification);
30  
31    /**
32     * 创建一个新规范，作为本规范的“非（NOT）”操作的结果
33     * @return 一个新规范。
34     */
35    Specification<T> not();
36  }
