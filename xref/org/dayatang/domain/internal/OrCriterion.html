3   import java.util.ArrayList;
4   import java.util.List;

6   import org.apache.commons.lang3.StringUtils;
7   import org.apache.commons.lang3.builder.EqualsBuilder;
8   import org.apache.commons.lang3.builder.HashCodeBuilder;
9   import org.dayatang.domain.NamedParameters;
10  import org.dayatang.domain.QueryCriterion;
11  import org.dayatang.utils.Assert;
12  
13  
14  /**
15   * 代表两个或两个以上的查询条件OR操作结果的查询条件
16   * @author yyang
17   */
18  public class OrCriterion extends AbstractCriterion {
19  
20      private final List<QueryCriterion> criterions;
21  
22      /**
23       * 根据多个查询条件创建OR查询条件。创建过程中会去除为null或EmptyCriterion的查询条件。如果
24       * 剩余查询条件不足两个，则抛出异常。
25       * @param criterions 要用来执行OR操作的查询条件
26       */
27      public OrCriterion(QueryCriterion... criterions) {
28          Assert.notNull(criterions, "Criterions to \"OR\" is null!");
29          this.criterions = removeNullOrEmptyCriterion(criterions);
30          Assert.isTrue(criterions.length > 1, "At least two query criterions required!");
31      }
32  
33      /**
34       * 获得要用来执行OR操作的查询条件
35       * @return 要用来执行AND操作的查询条件，去除了Null和EmptyCriterion类型的元素。
36       */
37      public List<QueryCriterion> getCriterons() {
38          return criterions;
39      }
40  
41      @Override
42  	public String toQueryString() {
43          List<String> subCriterionsStr = new ArrayList<String>();
44          for (QueryCriterion each : getCriterons()) {
45              subCriterionsStr.add(each.toQueryString());
46          }
47  		return "(" + StringUtils.join(subCriterionsStr, " or ") + ")";
48  	}
49  
50  	@Override
51  	public NamedParameters getParameters() {
52  		NamedParameters result = NamedParameters.create();
53          for (QueryCriterion each : getCriterons()) {
54          	result.add(each.getParameters());
55          }
56  		return result;
57  	}
58  
59  	/**
60       * 判断等价性
61       * @param other 要用来判等的另一个对象
62       * @return 如果当前对象和other等价，则返回true，否则返回false
63       */
64      @Override
65      public boolean equals(final Object other) {
66          if (this == other) {
67              return true;
68          }
69          if (!(other instanceof OrCriterion)) {
70              return false;
71          }
72          OrCriterion that = (OrCriterion) other;
73          return new EqualsBuilder()
74                  .append(this.getCriterons(), that.getCriterons())
75                  .isEquals();
76      }
77  
78      /**
79       * 计算哈希值
80       * @return 当前对象实例的哈希值
81       */
82      @Override
83      public int hashCode() {
84          return new HashCodeBuilder(17, 37).append(getCriterons()).toHashCode();
85      }
86  }
