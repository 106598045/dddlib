4   import org.dayatang.utils.Slf4JLogger;
5   
6   import java.io.IOException;
7   import java.io.InputStream;
8   import java.net.MalformedURLException;
9   import java.net.URL;
10  import java.util.Date;
11  import java.util.Hashtable;
12  import java.util.Properties;
13  
14  /**
15   * <P>ConfigurationUrlImpl为读取远程配置文件的工具类，一个实例大概对应了一个远程配置文件，具体配置大致采用
16   * ConfigurationUrlImpl.getXxx(key)的方式读取。</P>
17   * <P>每个配置项用key --> value 的方式组织，推荐采用点分字符串的方式编制key部分。 usePrefix()激活
18   * 配置项前缀功能，你可以通过usePrefix("xxx.xxx")设置某个具体实例的前缀。</P>
19   * <P>前缀的作用在于减少复杂性，如果我们在配置文件里有org.dayatang.smbserverhost=smbserverhost.com
20   * 这一项，并且不更改默认前缀的话，getXxx("smbserverhost")和get("org.dayatang.smbserverhost")
21   * 将会返回同样的结果。</P>
22   * <P>配置文件的格式符合标准的java属性文件格式，采用UTF8的编码方式，支持中文，不需native2ascii。</P>
23   * <P>注意：为了避免日期格式的转换等复杂问题，日期是转化为long类型的数据保存的（采用date.getTime()方法）。</P>
24   * 
25   * @author yyang (<a href="mailto:gdyangyu@gmail.com">gdyangyu@gmail.com</a>)
26   */
27  public class ConfigurationUrlImpl extends AbstractConfiguration {
28  	private static final Slf4JLogger LOGGER = Slf4JLogger.getLogger(ConfigurationUrlImpl.class);
29  	private PropertiesFileUtils pfu = new PropertiesFileUtils("utf-8");
30  	private URL url;
31  	
32  	public static ConfigurationUrlImpl fromUrl(final String url) {
33  		try {
34  			return new ConfigurationUrlImpl(new URL(url));
35  		} catch (MalformedURLException e) {
36  			e.printStackTrace();
37  			throw new IllegalStateException("URL " + url + " is malformed!" , e);
38  		}
39  	}
40  	
41  	public static ConfigurationUrlImpl fromUrl(final URL url) {
42  		return new ConfigurationUrlImpl(url);
43  	}
44  
45  	private ConfigurationUrlImpl(final URL url) {
46  		this.url = url;
47  		load();
48  	}
49  
50  	@Override
51  	public Properties getProperties() {
52  		return pfu.unRectifyProperties(getHashtable());
53  	}
54  
55  	@Override
56  	public void load() {
57  		hTable = new Hashtable<String, String>();
58  		Properties props = new Properties();
59  		InputStream in = null;
60  		try {
61  			in = url.openStream();
62  			props.load(in);
63  			hTable = pfu.rectifyProperties(props);
64  			LOGGER.debug("Load configuration from {} at {}", url, new Date());
65  		} catch (IOException e) {
66  			throw new ConfigurationException("Cannot load config file: " + url, e);
67  		} finally {
68  			if (in != null) {
69  				try {
70  					in.close();
71  				} catch (IOException e) {
72  					throw new ConfigurationException("Cannot close input stream.", e);
73  				}
74  			}
75  		}
76  	}
77  
78  	@Override
79  	public String toString() {
80  		return getClass().getSimpleName() + "{" + url + "}";
81  	}
82  }
