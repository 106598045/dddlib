9   import java.util.Map;
10  import java.util.Properties;

12  /**
13   * <P>ConfigurationDbImpl为读取/回写配置信息的工具类，并将配置信息写入数据库， 具体配置大致采用
14   * ConfigurationFileImpl.getXxx(key)的方式读取。</P>
15   * <P>每个配置项用key --> value 的方式组织，推荐采用点分字符串的方式编制key部分。 usePrefix()激活
16   * 配置项前缀功能，你可以通过usePrefix("xxx.xxx")设置某个具体实例的前缀。</P>
17   * <P>前缀的作用在于减少复杂性，如果我们在配置文件里有org.dayatang.smbserverhost=smbserverhost.com
18   * 这一项，并且不更改默认前缀的话，getXxx("smbserverhost")和get("org.dayatang.smbserverhost")
19   * 将会返回同样的结果。</P>
20   * <P>配置文件的格式符合标准的java属性文件格式，采用UTF8的编码方式，支持中文，不需native2ascii。</P>
21   * <P>注意：为了避免日期格式的转换等复杂问题，日期是转化为long类型的数据保存的（采用date.getTime()方法）。</P>
22   * 
23   * @author yyang (<a href="mailto:gdyangyu@gmail.com">gdyangyu@gmail.com</a>)
24   */
25  public class ConfigurationDbImpl extends AbstractConfiguration implements WritableConfiguration {
26  	
27  	private static final Slf4JLogger LOGGER = Slf4JLogger.getLogger(ConfigurationDbImpl.class);
28  	
29  	private ConfigurationDbUtils dbUtils;
30  	private static final String DEFAULT_TABLE_NAME = "SYS_CONFIG";
31  	private static final String DEFAULT_KEY_COLUMN = "KEY_COLUMN";
32  	private static final String DEFAULT_VALUE_COLUMN = "VALUE_COLUMN";
33  	
34  	
35  	public ConfigurationDbImpl(DataSource dataSource) {
36  		this(dataSource, DEFAULT_TABLE_NAME, DEFAULT_KEY_COLUMN, DEFAULT_VALUE_COLUMN);
37  	}
38  
39  	public ConfigurationDbImpl(DataSource dataSource, String tableName, String keyColumn, String valueColumn) {
40  		Assert.notNull(dataSource, "Data source is null!");
41  		Assert.notEmpty(tableName, "Table name is blank!");
42  		Assert.notEmpty(tableName, "Key column is blank!");
43  		Assert.notEmpty(tableName, "Value column is blank!");
44  		dbUtils = new ConfigurationDbUtils(dataSource, tableName, keyColumn, valueColumn);
45  	}
46  
47  	//从数据库中取得配置项，更新当前内存中的配置值。
48  	@Override
49  	public void load() {
50  		hTable = dbUtils.load();
51  		LOGGER.debug("Configuration info loaded from database at {}", new Date());
52  	}
53  
54  	/* (non-Javadoc)
55  	 * @see org.dayatang.configuration.WritableConfiguration#save()
56  	 */
57  	@Override
58  	public void save() {
59  		dbUtils.save(hTable);
60  	}
61  }
