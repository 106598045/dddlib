17  package org.dayatang.querychannel.query;
18  
19  import org.dayatang.domain.BaseQuery;
20  import org.dayatang.querychannel.ChannelQuery;
21  import java.util.List;
22  import org.dayatang.domain.Entity;
23  import org.dayatang.domain.EntityRepository;
24  import org.dayatang.domain.SqlQuery;
25  import org.dayatang.utils.Page;
26  import org.dayatang.utils.Assert;
27  
28  /**
29   * 通道查询的SQL实现
30   * @author yyang
31   */
32  public class ChannelSqlQuery extends ChannelQuery<ChannelSqlQuery> {
33      
34      private final SqlQuery query;
35  
36      public ChannelSqlQuery(EntityRepository repository, String sql) {
37          super(repository);
38          Assert.notBlank(sql, "SQL must be set!");
39          query = new SqlQuery(repository, sql);
40          setQuery(query);
41      }
42  
43      /**
44       * 设置查询的结果实体类型。注意setResultEntityClass()和addScalar()是互斥的，
45       * 分别适用于查询结果是实体和标量两种情形
46       * @param resultEntityClass 要设置的查询结果类型
47       * @return 该对象本身
48       */
49      public ChannelSqlQuery setResultEntityClass(Class<? extends Entity> resultEntityClass) {
50          query.setResultEntityClass(resultEntityClass);
51          return this;
52      }
53  
54      @Override
55      public <T> List<T> list() {
56          return query.list();
57      }
58  
59      @Override
60      public <T> Page<T> pagedList() {
61          return new Page<T>(query.getFirstResult(), queryResultCount(), 
62                  query.getMaxResults(), (List<T>) query.list());
63      }
64  
65      @Override
66      public <T> T singleResult() {
67          return (T) query.singleResult();
68      }
69  
70      @Override
71      protected String getQueryString() {
72          return query.getSql();
73      }
74  
75      @Override
76      protected BaseQuery createBaseQuery(String queryString) {
77          return repository.createSqlQuery(queryString);
78      }
79  
80  
81  }
