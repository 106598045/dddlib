16  package org.dayatang.persistence.jpa;
17  
18  import org.dayatang.domain.InstanceFactory;
19  import javax.persistence.EntityManager;
20  import javax.persistence.EntityManagerFactory;
21  import org.dayatang.domain.IocInstanceNotFoundException;
22  
23  /**
24   * JPA 实体管理器提供者。如果当前线程中尚未存在entityManager线程变量，则从IoC容器中获取一个并存入当前线程，
25   * 如果当前线程已经存在entityManager线程变量，直接返回。
26   * <p>
27   * 本类的存在，主要是为了在当前线程中，每次请求都返回相同的entityManager对象。避免事务和“会话已关闭”问题。
28   *
29   * @author yyang
30   */
31  public class EntityManagerProvider {
32  
33      private EntityManagerFactory entityManagerFactory;
34  
35      private final ThreadLocal<EntityManager> entityManagerHolder = new ThreadLocal<EntityManager>();
36  
37      public EntityManagerProvider() {
38          entityManagerFactory = InstanceFactory.getInstance(EntityManagerFactory.class);
39      }
40  
41      public EntityManagerProvider(EntityManagerFactory entityManagerFactory) {
42          this.entityManagerFactory = entityManagerFactory;
43      }
44  
45      public EntityManagerProvider(EntityManager entityManager) {
46          entityManagerHolder.set(entityManager);
47      }
48  
49      public EntityManager getEntityManager() {
50          EntityManager result = entityManagerHolder.get();
51          if (result != null && result.isOpen()) {
52              return result;
53          }
54          result = getEntityManagerFromIoC();
55          entityManagerHolder.set(result);
56          return result;
57      }
58  
59      public EntityManager getEntityManagerFromIoC() {
60          try {
61              return InstanceFactory.getInstance(EntityManager.class);
62          } catch (IocInstanceNotFoundException e) {
63              if (entityManagerFactory == null) {
64                  entityManagerFactory = InstanceFactory.getInstance(EntityManagerFactory.class);
65              }
66              return entityManagerFactory.createEntityManager();
67          }
68      }
69  }
