4   import com.google.inject.*;
5   import com.google.inject.name.Names;
6   import org.dayatang.domain.InstanceProvider;
7   
8   import java.lang.annotation.Annotation;
9   
10  /**
11   * 实例提供者接口的Google Guice实现。
12   *
13   * @author yyang (<a href="mailto:gdyangyu@gmail.com">gdyangyu@gmail.com</a>)
14   */
15  public class GuiceInstanceProvider implements InstanceProvider {
16      private final Injector injector;
17  
18      /**
19       * 以一批guice模块初始化guice实例提供者。
20       *
21       * @param modules 一或多个guice模块
22       */
23      public GuiceInstanceProvider(Module... modules) {
24          injector = Guice.createInjector(Stage.PRODUCTION, modules);
25      }
26  
27      /**
28       * 从Injector生成GuiceProvider
29       *
30       * @param injector
31       */
32      public GuiceInstanceProvider(Injector injector) {
33          this.injector = injector;
34      }
35  
36  
37      /**
38       * 根据类型获取对象实例。返回的对象实例所属的类是T或它的实现类或子类。如果找不到该类型的实例则返回null。
39       * @param <T> 类型参数
40       * @param beanType 实例的类型
41       * @return 指定类型的实例。
42       */
43      @Override
44      public <T> T getInstance(Class<T> beanType) {
45          try {
46              return injector.getInstance(beanType);
47          } catch (com.google.inject.ConfigurationException e) {
48              return null;
49          }
50      }
51  
52      /**
53       * 根据类型和名称获取对象实例。如果找不到该类型的实例则返回null。
54       * 假如有两个类MyService1和MyService2都实现了接口Service，而在Guice模块中这样注册：
55       * <pre>
56       * binder.bind(Service.class).to(MyService1.class)
57       * binder.bind(Service.class).annotatedWith(Names.named("service2")).to(MyService2.class)
58       * </pre>
59       * 那么getInstance(Service.class, "service2")将返回MyService2的实例。
60       *
61       * @param <T> 类型参数
62       * @param beanName 实现类在容器中配置的名字
63       * @param beanType 实例的类型
64       * @return 指定类型的实例。
65       */
66      @Override
67      public <T> T getInstance(Class<T> beanType, String beanName) {
68          Key<T> key = Key.get(beanType, Names.named(beanName));
69          try {
70              return injector.getInstance(key);
71          } catch (com.google.inject.ConfigurationException e) {
72              return null;
73          }
74      }
75  
76      /**
77       * 根据类型和Annotation获取对象实例。如果找不到该类型的实例则返回null。
78       * 假如有两个类MyService1和MyService2都实现了接口Service，其中MyService2标记为@MyAnnotation,
79       * 同时MyAnnotation标记为@BindingAnnotation @Retention(RetentionPolicy.RUNTIME) ,
80       * 而在Guice模块中这样注册：
81       * binder.bind(Service.class).to(MyService1.class)
82       * binder.bind(Service.class).annotatedWith(MyAnnotation.class).to(MyService2.class)
83       * 那么getInstance(Service.class, MyAnnotation.class)将返回MyService2的实例。
84       *
85       * @param <T> 类型参数
86       * @param beanType 实例的类型
87       * @param annotationType 实现类的annotation类型
88       * @return 指定类型的实例。
89       */
90      @Override
91      public <T> T getInstance(Class<T> beanType, Class<? extends Annotation> annotationType) {
92          Key<T> key = Key.get(beanType, annotationType);
93          try {
94              return injector.getInstance(key);
95          } catch (com.google.inject.ConfigurationException e) {
96              return null;
97          }
98      }
99  }
