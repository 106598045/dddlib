3   import org.apache.commons.beanutils.BeanUtils;
4   import org.slf4j.Logger;
5   import org.slf4j.LoggerFactory;
6   
7   import javax.sql.DataSource;
8   import java.lang.reflect.InvocationTargetException;
9   import java.sql.Driver;
10  import java.util.Map;
11  import java.util.Properties;
12  
13  public abstract class AbstractDataSourceBuilder implements DataSourceBuilder {
14  
15  	/**
16  	 * 
17  	 */
18  	private static final long serialVersionUID = -5389391685137579724L;
19  
20  	private static final Logger LOGGER = LoggerFactory.getLogger(AbstractDataSourceBuilder.class);
21  
22  	protected abstract Class<? extends DataSource> getDataSourceClass();
23  
24  	protected abstract String getPasswordProperty();
25  
26  	protected abstract String getUserProperty();
27  
28  	protected abstract String getJdbcUrlProperty();
29  
30  	protected abstract String getDriverProperty();
31  
32  	protected abstract Class<? extends Driver> getDriverClass(String url);
33  
34  	protected abstract String getJdbcUrl(String url);
35  
36  	protected abstract Properties getDefaultProp();
37  
38  	@Override
39  	public DataSource buildDataSource(Properties prop) {
40  		if (prop == null) {
41  			throw new IllegalArgumentException("datasource's prop is null.");
42  		}
43  
44  		DataSource dataSource = null;
45  		try {
46  			dataSource = getDataSourceClass().newInstance();
47  			debug("准备为【{}】构建数据源，prop=【{}】", new Object[] { getDataSourceClass(), prop });
48  			// dataSource = new ComboPooledDataSource();
49  			assembleProps(dataSource, prop);
50  		} catch (Exception ex) {
51  			ex.printStackTrace();
52  		}
53  		return dataSource;
54  	}
55  
56  	@Override
57  	public DataSource buildDataSource(String url, String username, String password, Properties prop) {
58  		DataSource dataSource = null;
59  		try {
60  			dataSource = getDataSourceClass().newInstance();
61  			debug("准备为【{}】构建数据源，url=【{}】，username=【{}】，password=【****】", new Object[] { getDataSourceClass(), url, username });
62  			// dataSource = new ComboPooledDataSource();
63  			assembleBasicProp(dataSource, url, username, password);
64  
65  			if (prop == null) {
66  				assembleProps(dataSource, getDefaultProp());
67  			} else {
68  				assembleProps(dataSource, prop);
69  			}
70  
71  		} catch (Exception ex) {
72  			ex.printStackTrace();
73  		}
74  		return dataSource;
75  	}
76  
77  	protected void assembleBasicProp(DataSource dataSource, String url, String username, String password) throws IllegalAccessException,
78  			InvocationTargetException {
79  		debug("正在为数据源【{}】组装基础属性driverClass=【{}】，jdbcUrl=【{}】，user=【{}】，password=【****】", new Object[] { getDataSourceClass(),
80  				getDriverClass(url).getName(), getJdbcUrl(url), username });
81  
82  		BeanUtils.setProperty(dataSource, getDriverProperty(), getDriverClass(url).getName());
83  		BeanUtils.setProperty(dataSource, getJdbcUrlProperty(), getJdbcUrl(url));
84  		BeanUtils.setProperty(dataSource, getUserProperty(), username);
85  		BeanUtils.setProperty(dataSource, getPasswordProperty(), password);
86  	}
87  
88  	private void assembleProps(DataSource dataSource, Properties properties) throws IllegalAccessException, InvocationTargetException {
89  		if (properties == null) {
90  			return;
91  		}
92  		for (Map.Entry<Object, Object> each : properties.entrySet()) {
93  			debug("正在为数据源【{}】组装可选属性【{}】=【{}】", new Object[] { dataSource, each.getKey(), each.getValue() });
94  			BeanUtils.setProperty(dataSource, (String) each.getKey(), each.getValue());
95  		}
96  	}
97  
98  	private void debug(String message, Object... params) {
99  		if (LOGGER.isDebugEnabled()) {
100 			LOGGER.debug(message, params);
101 		}
102 	}
103 }
