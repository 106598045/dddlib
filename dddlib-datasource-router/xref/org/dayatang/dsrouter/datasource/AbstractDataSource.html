16  public abstract class AbstractDataSource implements DataSource {
17  
18  	/** Logger available to subclasses */
19  	protected final Log logger = LogFactory.getLog(getClass());
20  
21  	/**
22  	 * Returns 0, indicating the default system timeout is to be used.
23  	 */
24  	public int getLoginTimeout() throws SQLException {
25  		return 0;
26  	}
27  
28  	/**
29  	 * Setting a login timeout is not supported.
30  	 */
31  	public void setLoginTimeout(int timeout) throws SQLException {
32  		throw new UnsupportedOperationException("setLoginTimeout");
33  	}
34  
35  	/**
36  	 * LogWriter methods are not supported.
37  	 */
38  	public PrintWriter getLogWriter() {
39  		throw new UnsupportedOperationException("getLogWriter");
40  	}
41  
42  	/**
43  	 * LogWriter methods are not supported.
44  	 */
45  	public void setLogWriter(PrintWriter pw) throws SQLException {
46  		throw new UnsupportedOperationException("setLogWriter");
47  	}
48  
49  	// ---------------------------------------------------------------------
50  	// Implementation of JDBC 4.0's Wrapper interface
51  	// ---------------------------------------------------------------------
52  
53  	@SuppressWarnings("unchecked")
54  	public <T> T unwrap(Class<T> iface) throws SQLException {
55  
56  		if (iface == null) {
57  			throw new IllegalArgumentException(
58  					"Interface argument must not be null");
59  		}
60  
61  		if (!DataSource.class.equals(iface)) {
62  			throw new SQLException(
63  					"DataSource of type ["
64  							+ getClass().getName()
65  							+ "] can only be unwrapped as [javax.sql.DataSource], not as ["
66  							+ iface.getName());
67  		}
68  		return (T) this;
69  	}
70  
71  	public boolean isWrapperFor(Class<?> iface) throws SQLException {
72  		return DataSource.class.equals(iface);
73  	}
74  
75  }
